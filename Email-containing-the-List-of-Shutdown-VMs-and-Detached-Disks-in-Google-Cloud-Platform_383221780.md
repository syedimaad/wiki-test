1.  **/mnt/vol1/gcp_mail/email.sh** - Path of the script  
    `email.sh`

    <div class="code panel pdl" style="border-width: 1px;">

    <div class="codeContent panelContent pdl">

    ``` syntaxhighlighter-pre
    #!/bin/bash

    > /mnt/vol1/gcp_mail/email

    projects=$(cat /mnt/vol1/gcp_mail/all_projects)

    for project in ${projects[@]}
    do

    project_name=$project
    shutdown_vms=$(gcloud compute instances list --filter="status:TERMINATED" --project ${project_name} | awk 'NR>1 {print $1}')
    detached_disks=$(gcloud compute disks list --filter="-users:*" --project ${project_name} | awk 'NR>1 {print $1}')
    subject="List of Shutdown VMs and Detached Disks in Google Cloud Platform"
    email_body="Project: ${project_name}.

    Shutdown VMs:
    ${shutdown_vms}

    Detached Disks:
    ${detached_disks}
    "
    echo "${email_body}" >> /mnt/vol1/gcp_mail/email

    done

    cat /mnt/vol1/gcp_mail/email | mailx -s "${subject}" -r "dh-devops@verse.in" dh-devops@verse.
    ```

    </div>
